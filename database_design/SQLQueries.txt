"number of orders per week" - Chosen for TA Demo
(Special Query 1)
SELECT EXTRACT(WEEK FROM Date) AS week_number, COUNT(OrderID) AS number_of_orders
FROM orders
GROUP BY week_number
ORDER BY week_number;

"total sales per hour of the day"
(Special Query 2)
SELECT EXTRACT(HOUR FROM Time) AS hour_of_day, COUNT(OrderID) AS number_of_orders, SUM(TotalCost) AS total_sales
FROM orders
GROUP BY hour_of_day
ORDER BY hour_of_day;

"top 10 days with highest sales" - Chosen for TA Demo
(Special Query 3)
SELECT Date, SUM(TotalCost) AS daily_sales
FROM orders
GROUP BY Date
ORDER BY daily_sales DESC
LIMIT 10;

"total revenue across all orders" - Chosen for TA Demo
(Normal Query 1)
SELECT SUM(TotalCost) AS total_revenue
FROM orders;

"total number of customers"
(Normal Query 2)
SELECT COUNT(CustomerID) AS total_customers
FROM customers;

"total number of orders" - Chosen for TA Demo
(Normal Query 3)
SELECT COUNT(OrderID) AS total_orders
FROM orders;

"average order value per customer"
(Normal Query 4)
SELECT AVG(TotalCost) AS average_order_value
FROM orders;

"sales by menu item category"
(Normal Query 5)
SELECT mi.Category, SUM(oi.Price) AS category_sales
FROM menu_items mi
JOIN order_items oi ON mi.Name = oi.Drink
GROUP BY mi.Category
ORDER BY category_sales DESC;

"top 10 customers by order count"
(Normal Query 6)
SELECT c.Name, COUNT(o.OrderID) AS number_of_orders
FROM customers c
JOIN orders o ON c.CustomerID = o.CustomerID
GROUP BY c.Name
ORDER BY number_of_orders DESC
LIMIT 10;

"total sales per employee"
(Normal Query 7)
SELECT e.Name, SUM(o.TotalCost) AS total_revenue
FROM employees e
JOIN orders o ON e.EmployeeID = o.EmployeeID
GROUP BY e.Name
ORDER BY total_revenue DESC;

"most popular toppings"
(Normal Query 8)
SELECT Toppings, COUNT(*) AS times_ordered
FROM order_items
WHERE Toppings IS NOT NULL
GROUP BY Toppings
ORDER BY times_ordered DESC
LIMIT 5;

"average number of items per order"
(Normal Query 9)
SELECT AVG(item_count) AS average_items_per_order
FROM (
    SELECT OrderID, COUNT(ItemID) AS item_count
    FROM order_items
    GROUP BY OrderID
) AS order_counts;
